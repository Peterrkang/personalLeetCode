Two Pointers

When: sorted array/string; pair-sum, dedup, partition, palindrome, in-place merge.

How: two indices move toward each other (or same direction) based on a comparison, shrinking the search space each step.

Big-O: O(n) time, O(1) space.

function twoPointers(arr, target) {
  let left = 0, right = arr.length - 1;
  while (left < right) {
    const total = arr[left] + arr[right];
    if (total === target) return [left, right];
    else if (total < target) left++;   // need bigger -> move left up
    else right--;                      // need smaller -> move right down
  }
  return [];
}

Sliding Window

When: contiguous subarray/substring; "longest/shortest/at most K distinct/sum equals".

How: expand right to grow the window; shrink from left while the window violates the constraint; record the best valid window.

Big-O: O(n) time, O(k) space.

function slidingWindow(s, isValid) {
  let left = 0, best = 0;
  const state = new Map();              // window state: counts, sum, etc.
  for (let right = 0; right < s.length; right++) {
    add(state, s[right]);               // 1. include s[right]
    while (!isValid(state)) {           // 2. shrink while invalid
      remove(state, s[left]);
      left++;
    }
    best = Math.max(best, right - left + 1);  // 3. record valid window
  }
  return best;
}

Prefix Sum

When: many range-sum / subarray-sum queries; "subarray sums to K".

How: precompute running sums; sum of range (i, j] = prefix[j] - prefix[i]. Pair with a hash map to count subarrays in one pass.

Big-O: brute force nested loops O(n²) → one pass with hash map O(n).

function subarraySumEqualsK(nums, k) {
  let count = 0, prefix = 0;
  const seen = new Map([[0, 1]]);       // prefixSum -> times seen
  for (const n of nums) {
    prefix += n;
    count += seen.get(prefix - k) || 0; // complement seen before?
    seen.set(prefix, (seen.get(prefix) || 0) + 1);
  }
  return count;
}

Binary Search

When: sorted input, OR a monotonic "yes/no" answer space ("min/max value that satisfies X").

How: halve the search range each step using a predicate. The hard part is defining the predicate and the boundary update.

Big-O: O(log n).

function binarySearch(lo, hi, condition) {
  // smallest x in [lo, hi] where condition(x) is true
  while (lo < hi) {
    const mid = Math.floor((lo + hi) / 2);
    if (condition(mid)) hi = mid;       // answer is mid or to the left
    else lo = mid + 1;                  // answer is strictly right
  }
  return lo;
}

Fast & Slow Pointers (Floyd's)

When: linked list cycle detection, find middle, find cycle start.

How: slow moves 1 step, fast moves 2; they meet inside a cycle.

Big-O: O(n) time, O(1) space.

function hasCycle(head) {
  let slow = head, fast = head;
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
    if (slow === fast) return true;
  }
  return false;
}

DFS (recursive)

When: trees/graphs/grids; explore a path fully before backtracking; connected components, flood fill.

How: visit node, mark seen, recurse on neighbors.

Big-O: O(V + E); grids O(rows·cols).

function dfs(node, visited, graph) {
  if (visited.has(node)) return;
  visited.add(node);
  // process(node)
  for (const next of graph.get(node) || []) {
    dfs(next, visited, graph);
  }
}

BFS (level-order / shortest path in unweighted graph)

When: shortest path in unweighted graph or grid; level-by-level traversal.

How: queue-based; process one full "level" per outer loop to track distance.

Big-O: O(V + E).

function bfs(start, graph) {
  const visited = new Set([start]);
  let queue = [start];
  let level = 0;
  while (queue.length) {
    const next = [];
    for (const node of queue) {         // one whole level
      // process(node, level)
      for (const nb of graph.get(node) || []) {
        if (!visited.has(nb)) {
          visited.add(nb);
          next.push(nb);
        }
      }
    }
    queue = next;
    level++;
  }
  return level;
}

Backtracking (place → explore → remove)

When: generate all subsets/permutations/combinations; constraint satisfaction (N-Queens, Sudoku).

How: the readable 3-step gist — place a choice, explore deeper recursively, remove the choice to try the next.

Big-O: exponential (the search tree); prune aggressively.

function backtrack(path, choices, result) {
  if (isComplete(path)) {
    result.push([...path]);             // record a copy
    return;
  }
  for (const choice of choices) {
    if (!isValid(choice, path)) continue;
    path.push(choice);                  // PLACE
    backtrack(path, nextChoices(choices, choice), result);  // EXPLORE
    path.pop();                         // REMOVE
  }
}

Dynamic Programming — Top-Down (Memoization)

When: overlapping subproblems + optimal substructure; "number of ways", "min/max cost", "can you reach".

How: recurse on a clear state; cache results so each state is computed once.

Big-O: O(#states · work per state).

function solve(n) {
  const memo = new Map();
  function dp(state) {
    if (baseCase(state)) return baseValue;
    if (memo.has(state)) return memo.get(state);
    let best = Infinity;                // or -Infinity / 0 depending on goal
    for (const next of transitions(state)) {
      best = Math.min(best, cost(state, next) + dp(next));
    }
    memo.set(state, best);
    return best;
  }
  return dp(initialState);
}

Dynamic Programming — Bottom-Up (Tabulation)

When: same problems, when you want iterative + easy space optimization.

How: fill a table from base cases up to the target; often reducible to O(1)/O(n) rows.

function dpTable(n) {
  const dp = new Array(n + 1).fill(0);
  dp[0] = baseValue;
  for (let i = 1; i <= n; i++) {
    for (const choice of choices(i)) {
      dp[i] = best(dp[i], dp[i - choice] + 1);
    }
  }
  return dp[n];
}

Heap / Priority Queue (Top-K)

When: "K largest/smallest/closest", merge K sorted lists, schedule by priority.

How: keep a heap of size K. JS has no built-in heap, so here's a compact min-heap; push, and pop the smallest when size exceeds K.

Big-O: O(n log k).

class MinHeap {
  constructor() { this.h = []; }
  size() { return this.h.length; }
  peek() { return this.h[0]; }
  push(v) {
    const h = this.h;
    h.push(v);
    let i = h.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (h[p] <= h[i]) break;
      [h[p], h[i]] = [h[i], h[p]];
      i = p;
    }
  }
  pop() {
    const h = this.h;
    const top = h[0], last = h.pop();
    if (h.length) {
      h[0] = last;
      let i = 0;
      while (true) {
        let l = 2 * i + 1, r = 2 * i + 2, s = i;
        if (l < h.length && h[l] < h[s]) s = l;
        if (r < h.length && h[r] < h[s]) s = r;
        if (s === i) break;
        [h[s], h[i]] = [h[i], h[s]];
        i = s;
      }
    }
    return top;
  }
}

function topK(nums, k) {
  const heap = new MinHeap();
  for (const n of nums) {
    heap.push(n);
    if (heap.size() > k) heap.pop();    // drop smallest -> heap holds k largest
  }
  return heap.h;
}

Monotonic Stack

When: "next greater/smaller element", daily temperatures, largest rectangle in histogram.

How: keep the stack in increasing (or decreasing) order; when the new item breaks the order, pop and resolve those elements.

Big-O: O(n) — each element pushed/popped once.

function nextGreater(nums) {
  const res = new Array(nums.length).fill(-1);
  const stack = [];                     // holds indices, values decreasing
  for (let i = 0; i < nums.length; i++) {
    while (stack.length && nums[stack[stack.length - 1]] < nums[i]) {
      const j = stack.pop();
      res[j] = nums[i];                 // nums[i] is next greater for index j
    }
    stack.push(i);
  }
  return res;
}

Topological Sort (Kahn's / BFS)

When: ordering with dependencies — course schedule, build systems, task scheduling; DAGs.

How: repeatedly remove nodes with in-degree 0. If you can't remove all nodes, there's a cycle.

Big-O: O(V + E).

function topoSort(graph, indegree) {
  // graph: Map<node, node[]>, indegree: Map<node, number>
  const queue = [];
  for (const [node, deg] of indegree) if (deg === 0) queue.push(node);
  const order = [];
  while (queue.length) {
    const node = queue.shift();
    order.push(node);
    for (const next of graph.get(node) || []) {
      indegree.set(next, indegree.get(next) - 1);
      if (indegree.get(next) === 0) queue.push(next);
    }
  }
  return order.length === graph.size ? order : [];  // [] => cycle
}

Dijkstra (shortest path, non-negative weights)

When: weighted graph shortest path with non-negative edges.

How: greedily expand the closest unfinalized node using a min-heap of [distance, node]. (Reuses the MinHeap above, comparing on element [0].)

Big-O: O(E log V).

function dijkstra(graph, start) {
  // graph: Map<node, [neighbor, weight][]>
  const dist = new Map([[start, 0]]);
  const heap = new MinHeapBy((a, b) => a[0] - b[0]);
  heap.push([0, start]);                // [distanceSoFar, node]
  while (heap.size()) {
    const [d, node] = heap.pop();
    if (d > (dist.get(node) ?? Infinity)) continue;  // stale entry
    for (const [next, w] of graph.get(node) || []) {
      const nd = d + w;
      if (nd < (dist.get(next) ?? Infinity)) {
        dist.set(next, nd);
        heap.push([nd, next]);
      }
    }
  }
  return dist;
}

MinHeapBy is the same heap as above but takes a comparator (a, b) => number and replaces every h[x] <= h[y] / h[x] < h[y] check with cmp(a, b) <= 0 / < 0. Keep one generic comparator-based heap in your toolkit and reuse it everywhere.

Union-Find (Disjoint Set)

When: connected components, cycle detection in undirected graphs, Kruskal's MST, "are these in the same group".

How: each set has a representative root; find with path compression, union by rank keep it near O(1).

Big-O: ~O(α(n)) ≈ O(1) per op.

class UnionFind {
  constructor(n) {
    this.parent = Array.from({ length: n }, (_, i) => i);
    this.rank = new Array(n).fill(0);
  }
  find(x) {
    while (this.parent[x] !== x) {
      this.parent[x] = this.parent[this.parent[x]];  // path compression
      x = this.parent[x];
    }
    return x;
  }
  union(a, b) {
    let ra = this.find(a), rb = this.find(b);
    if (ra === rb) return false;        // already connected -> cycle
    if (this.rank[ra] < this.rank[rb]) [ra, rb] = [rb, ra];
    this.parent[rb] = ra;
    if (this.rank[ra] === this.rank[rb]) this.rank[ra]++;
    return true;
  }
}

Trie (Prefix Tree)

When: word dictionary, prefix search, autocomplete, "starts with".

How: each node holds children by character + an end-of-word flag.

Big-O: insert/search O(L), L = word length.

class TrieNode {
  constructor() { this.children = new Map(); this.isEnd = false; }
}

class Trie {
  constructor() { this.root = new TrieNode(); }
  insert(word) {
    let node = this.root;
    for (const ch of word) {
      if (!node.children.has(ch)) node.children.set(ch, new TrieNode());
      node = node.children.get(ch);
    }
    node.isEnd = true;
  }
  search(word, prefixOnly = false) {
    let node = this.root;
    for (const ch of word) {
      if (!node.children.has(ch)) return false;
      node = node.children.get(ch);
    }
    return prefixOnly ? true : node.isEnd;
  }
}

Merge Intervals

When: overlapping intervals — merge, insert, count meeting rooms.

How: sort by start; sweep, merging when the current start ≤ previous end.

Big-O: O(n log n) from the sort.

function mergeIntervals(intervals) {
  intervals.sort((a, b) => a[0] - b[0]);
  const merged = [];
  for (const [start, end] of intervals) {
    const last = merged[merged.length - 1];
    if (last && start <= last[1]) last[1] = Math.max(last[1], end);  // overlap
    else merged.push([start, end]);
  }
  return merged;
}

Quickselect (Kth element without full sort)

When: Kth largest/smallest, median, when full sorting is overkill.

How: partition like quicksort but recurse into only the side containing K.

Big-O: O(n) average, O(n²) worst (random pivot to avoid).

function quickselect(nums, k) {          // k-th smallest, 1-indexed
  function partition(lo, hi, pivotIdx) {
    const pivot = nums[pivotIdx];
    [nums[pivotIdx], nums[hi]] = [nums[hi], nums[pivotIdx]];
    let store = lo;
    for (let i = lo; i < hi; i++) {
      if (nums[i] < pivot) {
        [nums[store], nums[i]] = [nums[i], nums[store]];
        store++;
      }
    }
    [nums[hi], nums[store]] = [nums[store], nums[hi]];
    return store;
  }

  let lo = 0, hi = nums.length - 1;
  const target = k - 1;
  while (lo <= hi) {
    const p = partition(lo, hi, lo + Math.floor(Math.random() * (hi - lo + 1)));
    if (p === target) return nums[p];
    else if (p < target) lo = p + 1;
    else hi = p - 1;
  }
}

JS-specific interview reminders

  • No built-in heap/priority queue — bring your own (above).
  • array.shift() is O(n); for hot BFS loops use a level-swap (shown) or an index pointer.
  • Map/Set over {}/[] for hash needs: real O(1), any key type, predictable iteration order.
  • Math.floor((lo + hi) / 2) — JS division returns a float, so floor it.
  • Sort is lexicographic by default: always pass a comparator for numbers, (a, b) => a - b.
  • Recursion depth is limited (~10k+ frames); convert deep DFS/DP to iterative + explicit stack if you risk a stack overflow.
